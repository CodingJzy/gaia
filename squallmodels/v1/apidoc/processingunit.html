ProcessingUnit

A Processing Unit reprents anything that can compute. It can be a Docker container, or a simple Unix process. They are created, updated and deleted by the system as they come and go. You can only modify its tags. Processing Units use Network Access Policies to define which other Processing Units or External Services they can communicate with andFile Access Policies to define what File Paths they can use.

Code example

Retrieve all ProcessingUnits in a given Vulnerability

apoctl Go Python REST 
apoctl api list processingunits in vulnerability parent-id -n /namespace

package main

import (
  "github.com/aporeto-inc/manipulate"
  "github.com/aporeto-inc/manipulate/manipwebsocket"

  squallmodels "github.com/aporeto-inc/gaia/squallmodels/current/golang"
)

func main() {

  m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

  if err != nil {
      panic(err)
  }

  var objects squallmodels.ProcessingUnitsList

  parent := squallmodels.NewVulnerability()
  parent.SetIdentifier("parent-id")

  mctx := manipulate.NewContext()
  mctx.Parent = parent

  if err = m.RetrieveMany(mctx, &objects); err != nil {
    panic(err)
  }

  disconnect()
}
  
import os

from squallmodels import ProcessingUnit, Vulnerability
from pymanipulate.context import Context
from pymanipulate.maniphttp import HTTPManipulator, PyManipulateError


def main():
    """
    """
    certificate = ("cert.pem", "key.pem")
    manipulator = HTTPManipulator("url", "/namespace", "username", "password", None, certificate)

    obj = ProcessingUnit()
    objects = []

    parent = Vulnerability(ID="parent-id")
    mctx = Context(parent=parent)

    try:
        manipulator.retrieve_children(mctx, obj.identity(), objects)
    except PyManipulateError as e:
        print "Err: %s" % e.errors[0]
        os._exit(1)

    for obj in objects:
        print obj.to_dict()

    exit(0)


if __name__ == "__main__":
    main()
  
curl -X GET \
  -H "X-Namespace: namespace" \
  -H "Authorization: token" \
  http://<host>:<port>/vulnerabilities/id/processingunits
