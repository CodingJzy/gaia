NetworkAccessPolicy

Allows to define networking policies to allow or prevent processing units identitied by their tags to talk to other processing units or external services (also identified by their tags).

Attributes

filterable orderable
allowsTraffic boolean
AllowsTraffic if true, the flow will be accepted. Otherwise other actions like "logs" can still be done, but the traffic will be rejected.
annotation annotation
Annotation stores additional information about an entity
associatedTags tags_list
AssociatedTags are the list of tags attached to an entity

createTime time
CreatedTime is the time at which the object was created
filterable orderable
description string
Description is the description of the object. (format: free)
filterable orderable
destinationPorts ports_list
DestinationPorts contains the list of allowed ports and ranges.
filterable orderable
disabled boolean
Disabled defines if the propert is disabled.

EncryptionEnabled defines if the flow has to be encrypted.

Code example

Retrieve all NetworkAccessPolicies

apoctl Go Python REST 
apoctl api list networkaccesspolicies -n /namespace
  
package main

import (
  "github.com/aporeto-inc/manipulate/manipwebsocket"

  squallmodels "github.com/aporeto-inc/gaia/squallmodels/current/golang"
)

func main() {

  m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

  if err != nil {
      panic(err)
  }

  var objects squallmodels.NetworkAccessPoliciesList

  if err = m.RetrieveMany(nil, &objects); err != nil {
    panic(err)
  }

  disconnect()
}
  
import os

from squallmodels import NetworkAccessPolicy
from pymanipulate.maniphttp import HTTPManipulator, PyManipulateError


def main():
    """
    """
    certificate = ("cert.pem", "key.pem")
    manipulator = HTTPManipulator("url", "/namespace", "username", "password", None, certificate)

    obj = NetworkAccessPolicy()
    objects = []

    try:
        manipulator.retrieve_children(None, obj.identity(), objects)
    except PyManipulateError as e:
        print "Err: %s" % e.errors[0]
        os._exit(1)

    for obj in objects:
        print obj.to_dict()

    exit(0)


if __name__ == "__main__":
    main()
