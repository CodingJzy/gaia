AWSAccount

Allows to bind an AWS account to your Aporeto account to allow auto registration of enforcers running on EC2

Code example

Retrieve all AWSAccounts

apoctl Go Python REST 
apoctl api list awsaccounts -n /namespace

package main

import (
  "github.com/aporeto-inc/manipulate/manipwebsocket"

  vincemodels "github.com/aporeto-inc/gaia/vincemodels/current/golang"
)

func main() {

  m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

  if err != nil {
      panic(err)
  }

  var objects vincemodels.AWSAccountsList

  if err = m.RetrieveMany(nil, &objects); err != nil {
    panic(err)
  }

  disconnect()
}
  
import os

from vincemodels import AWSAccount
from pymanipulate.maniphttp import HTTPManipulator, PyManipulateError


def main():
    """
    """
    certificate = ("cert.pem", "key.pem")
    manipulator = HTTPManipulator("url", "/namespace", "username", "password", None, certificate)

    obj = AWSAccount()
    objects = []

    try:
        manipulator.retrieve_children(None, obj.identity(), objects)
    except PyManipulateError as e:
        print "Err: %s" % e.errors[0]
        os._exit(1)

    for obj in objects:
        print obj.to_dict()

    exit(0)


if __name__ == "__main__":
    main()

Create a new AWSAccount

apoctl Go Python REST 
apoctl api create awsaccounts -n /namespace \
  -k accesskeyid "accesskeyid" \
  -k region "region" \
  -k secretaccesskey "secretaccesskey"
package main

import (
    "github.com/aporeto-inc/manipulate/manipwebsocket"

    vincemodels "github.com/aporeto-inc/gaia/vincemodels/current/golang"
)

func main () {

    m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

    if err != nil {
        panic(err)
    }

    obj := vincemodels.NewAWSAccount()
    obj.AccessKeyID = "accesskeyid"
    obj.Region = "region"
    obj.SecretAccessKey = "secretaccesskey"

    if err = m.Create(nil, obj); err != nil {
        panic(err)
    }

    disconnect()
}
