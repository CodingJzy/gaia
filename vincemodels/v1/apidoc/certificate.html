Code example

Retrieve all Certificates

apoctl Go Python REST 
apoctl api list certificates -n /namespace
  
package main

import (
  "github.com/aporeto-inc/manipulate/manipwebsocket"

  vincemodels "github.com/aporeto-inc/gaia/vincemodels/current/golang"
)

func main() {

  m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

  if err != nil {
      panic(err)
  }

  var objects vincemodels.CertificatesList

  if err = m.RetrieveMany(nil, &objects); err != nil {
    panic(err)
  }

  disconnect()
}
  
import os

from vincemodels import Certificate
from pymanipulate.maniphttp import HTTPManipulator, PyManipulateError


def main():
    """
    """
    certificate = ("cert.pem", "key.pem")
    manipulator = HTTPManipulator("url", "/namespace", "username", "password", None, certificate)

    obj = Certificate()
    objects = []

    try:
        manipulator.retrieve_children(None, obj.identity(), objects)
    except PyManipulateError as e:
        print "Err: %s" % e.errors[0]
        os._exit(1)

    for obj in objects:
        print obj.to_dict()

    exit(0)


if __name__ == "__main__":
    main()

Create a new Certificate

apoctl Go Python REST 
apoctl api create certificates -n /namespace \
  -k commonname "commonname" \
  -k name "name" \
  -k passphrase "passphrase"
package main

import (
    "github.com/aporeto-inc/manipulate/manipwebsocket"

    vincemodels "github.com/aporeto-inc/gaia/vincemodels/current/golang"
)

func main () {

    m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

    if err != nil {
        panic(err)
    }

    obj := vincemodels.NewCertificate()
    obj.CommonName = "commonname"
    obj.Name = "name"
    obj.Passphrase = "passphrase"

    if err = m.Create(nil, obj); err != nil {
        panic(err)
    }

    disconnect()
}
