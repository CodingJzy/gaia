Retrieve all Plans in a given Plan

apoctl Go Python REST 
apoctl api list plans in plan parent-id -n /namespace

package main

import (
  "github.com/aporeto-inc/manipulate"
  "github.com/aporeto-inc/manipulate/manipwebsocket"

  vincemodels "github.com/aporeto-inc/gaia/vincemodels/current/golang"
)

func main() {

  m, disconnect, err := manipwebsocket.NewWebSocketManipulator("username", "token", "url", "namespace")

  if err != nil {
      panic(err)
  }

  var objects vincemodels.PlansList

  parent := vincemodels.NewPlan()
  parent.SetIdentifier("parent-id")

  mctx := manipulate.NewContext()
  mctx.Parent = parent

  if err = m.RetrieveMany(mctx, &objects); err != nil {
    panic(err)
  }

  disconnect()
}
  
import os

from vincemodels import Plan, Plan
from pymanipulate.context import Context
from pymanipulate.maniphttp import HTTPManipulator, PyManipulateError


def main():
    """
    """
    certificate = ("cert.pem", "key.pem")
    manipulator = HTTPManipulator("url", "/namespace", "username", "password", None, certificate)

    obj = Plan()
    objects = []

    parent = Plan(ID="parent-id")
    mctx = Context(parent=parent)

    try:
        manipulator.retrieve_children(mctx, obj.identity(), objects)
    except PyManipulateError as e:
        print "Err: %s" % e.errors[0]
        os._exit(1)

    for obj in objects:
        print obj.to_dict()

    exit(0)


if __name__ == "__main__":
    main()
  
curl -X GET \
  -H "X-Namespace: namespace" \
  -H "Authorization: token" \
  http://<host>:<port>/plans/id/plans
